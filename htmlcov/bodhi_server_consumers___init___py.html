18""" 

19fedora-messaging consumer. 

20 

21This module is responsible for consuming the messaging from the fedora-messaging bus. 

22It has the role to inspect the topics of the message and call the correct handler. 

23""" 

24from collections import namedtuple 

25import logging 

26 

27import fedora_messaging 

28 

29from bodhi.server import bugs, buildsys, initialize_db 

30from bodhi.server.config import config 

31from bodhi.server.consumers.automatic_updates import AutomaticUpdateHandler 

32from bodhi.server.consumers.signed import SignedHandler 

33from bodhi.server.consumers.greenwave import GreenwaveHandler 

34from bodhi.server.consumers.ci import CIHandler 

35 

36 

37log = logging.getLogger('bodhi') 

38 

39 

40HandlerInfo = namedtuple('HandlerInfo', ['topic_suffix', 'name', 'handler']) 

41 

42 

43class Consumer: 

44 """All Bodhi messages are received by this class's __call__() method.""" 

45 

46 def __init__(self): 

47 """Set up the database, build system, bug tracker, and handlers.""" 

48 log.info('Initializing Bodhi') 

49 initialize_db(config) 

50 buildsys.setup_buildsystem(config) 

51 bugs.set_bugtracker() 

52 

53 self.handler_infos = [ 

54 HandlerInfo('.buildsys.tag', "Signed", SignedHandler()), 

55 HandlerInfo('.buildsys.tag', 'Automatic Update', AutomaticUpdateHandler()), 

56 HandlerInfo('.greenwave.decision.update', 'Greenwave', GreenwaveHandler()), 

57 HandlerInfo('.ci.koji-build.test.running', 'CI', CIHandler()) 

58 ] 

59 

60 def __call__(self, msg: fedora_messaging.api.Message): # noqa: D401 

61 """ 

62 Callback method called by fedora-messaging consume. 

63 

64 Redirect messages to the correct handler using the 

65 message topic. 

66 

67 Args: 

68 msg: The message received from the broker. 

69 """ 

70 log.info(f'Received message from fedora-messaging with topic: {msg.topic}') 

71 

72 error_handlers_msgs = [] 

73 

74 for handler_info in self.handler_infos: 

75 if not msg.topic.endswith(handler_info.topic_suffix): 

76 continue 

77 log.debug(f'Passing message to the {handler_info.name} handler') 

78 try: 

79 handler_info.handler(msg) 

80 except Exception as e: 

81 log.exception(f'{str(e)}: Unable to handle message in {handler_info.name} handler: ' 

82 f'{msg}') 

83 error_handlers_msgs.append((handler_info.name, str(e))) 

84 

85 if error_handlers_msgs: 

86 error_msg = "Unable to (fully) handle message.\nAffected handlers:\n" 

87 for handler, exc in error_handlers_msgs: 

88 error_msg += f"\t{handler}: {exc}\n" 

89 error_msg += "Message:\n{msg}" 

90 raise fedora_messaging.exceptions.Nack(error_msg)
