29class BuildrootOverrideMessage(BodhiMessage): 

30 """Base class for the buildroot_override messages.""" 

31 

32 @property 

33 def build(self) -> BuildV1: 

34 """Return the build that was overridden.""" 

35 return BuildV1(self.body['override']['nvr']) 

36 

37 @property 

38 def submitter(self) -> UserV1: 

39 """Return the name of the submitter for the override.""" 

40 return UserV1(self.body['override']['submitter']['name']) 

41 

42 @property 

43 def summary(self) -> str: 

44 """ 

45 Return a short, human-readable representation of this message. 

46 

47 This should provide a short summary of the message, much like the subject line 

48 of an email. 

49 """ 

50 return self._summary_tmpl.format(submitter=self.submitter.name, build=self.build.nvr) 

51 

52 @property 

53 def url(self) -> str: 

54 """ 

55 Return a URL to the action that caused this message to be emitted. 

56 

57 Returns: 

58 A relevant URL. 

59 """ 

60 return f"https://bodhi.fedoraproject.org/overrides/{self.build.nvr}" 

61 

62 @property 

63 def packages(self) -> typing.Iterable[str]: 

64 """List of packages affected by the action that generated this message. 

65 

66 Returns: 

67 A list of affected package names. 

68 """ 

69 return [self.build.package] 

70 

71 @property 

72 def agent(self) -> str: 

73 """Return the agent's username for this message. 

74 

75 Returns: 

76 The agent's username. 

77 """ 

78 return self.submitter.name 

79 

80 @property 

81 def usernames(self) -> typing.List[str]: 

82 """ 

83 List of users affected by the action that generated this message. 

84 

85 Returns: 

86 A list of affected usernames. 

87 """ 

88 return [self.agent]

91class BuildrootOverrideTagV1(BuildrootOverrideMessage): 

92 """Sent when a buildroot override is added and tagged into the build root.""" 

93 

94 body_schema = { 

95 'id': f'{SCHEMA_URL}/v1/bodhi.buildroot_override.tag#', 

96 '$schema': 'http://json-schema.org/draft-04/schema#', 

97 'description': 'Schema for message sent when buildroot overrides are tagged', 

98 'type': 'object', 

99 'properties': { 

100 'override': { 

101 'type': 'object', 

102 'properties': { 

103 'nvr': { 

104 'type': 'string', 

105 'description': 'The NVR of the build that was overridden' 

106 }, 

107 'submitter': UserV1.schema(), 

108 }, 

109 'required': ['nvr', 'submitter'] 

110 } 

111 }, 

112 'required': ['override'], 

113 } 

114 

115 topic = "bodhi.buildroot_override.tag" 

116 _summary_tmpl = "{submitter} submitted a buildroot override for {build}"
