31def get_configfile() -> typing.Optional[str]: 

32 """ 

33 Return a path to a config file, if found. 

34 

35 Return the path to a config file, with a hierarchy of preferential paths. It searches first 

36 for development.ini if found. If not found, it will return /etc/bodhi/production.ini if it 

37 exists. Otherwise, it returns None. 

38 

39 Returns: 

40 The path of a config file, or None if no config file is found. 

41 """ 

42 configfile = None 

43 setupdir = os.path.join(os.path.dirname(os.path.dirname(__file__)), '..') 

44 for cfg in (os.path.join(setupdir, 'development.ini'), 

45 '/etc/bodhi/production.ini'): 

46 if os.path.exists(cfg): 

47 configfile = cfg 

48 break 

49 else: 

50 log.error("Unable to find configuration to load!") 

51 return configfile

54def _generate_dict_validator(value: str) -> dict: 

55 """Return a dict version of value. 

56 

57 This function accept a string with comma separated tuples in the form `key:value` 

58 and ensure it can be correctly imported as a dict. It will return a dict. 

59 If it cannot do that, it will raise ValueError. 

60 

61 Args: 

62 value: The value to be validated as a dict. 

63 Returns: 

64 The dict interpretation of value. 

65 Raises: 

66 ValueError: If value cannot be interpreted as a dict. 

67 """ 

68 if isinstance(value, str): 

69 values_list = [idx.strip() for idx in value.split(',') if idx.strip()] 

70 dictvalue = dict() 

71 for v in values_list: 

72 try: 

73 k, v = v.split(':') 

74 dictvalue[k.strip()] = v.strip() 

75 except Exception: 

76 raise ValueError(f'"{value}" cannot be interpreted as a dict.') 

77 value = dictvalue 

78 

79 if not isinstance(value, dict): 

80 raise ValueError(f'"{value}" cannot be interpreted as a dict.') 

81 

82 return value

85def _generate_list_validator( 

86 splitter: str = ' ', validator: typing.Callable[[typing.Any], typing.Any] = str) \ 

87 -> typing.Callable[[typing.Union[str, typing.List]], typing.Any]: 

88 """Return a function that takes a value and interprets it to be a list with the given splitter. 

89 

90 This function generates a function that can take a string and interpret it as a list by 

91 splitting with the given splitter string. Each element of the resulting list is then validated 

92 with the given validator. 

93 

94 Args: 

95 splitter: A string to use to split the input into a list. 

96 validator: A function to apply to each element of the list to validate it. 

97 Returns: 

98 A validator function that accepts an argument to be validated. 

99 """ 

100 def _validate_list(value: typing.Union[str, typing.List]) -> typing.List: 

101 """Validate that the value is a list or can be split into a list, and validate its elements. 

102 

103 This function will validate that the given value is a list, or it will use the splitter to 

104 turn it into a list. Once it is a list, it will use the validator on each element of the 

105 list. 

106 

107 Args: 

108 value: The list to be validated. 

109 Returns: 

110 The interpreted list. 

111 Raises: 

112 ValueError: If validator fails on any of the list's elements. 

113 """ 

114 if isinstance(value, str): 

115 value = [idx.strip() for idx in value.split(splitter) if idx.strip()] 

116 

117 if not isinstance(value, list): 

118 raise ValueError('"{}" cannot be interpreted as a list.'.format(value)) 

119 

120 # Run the validator on each element of the list. 

121 value = [validator(v) for v in value] 

122 

123 return value 

124 

125 return _validate_list

128def _validate_bool(value: typing.Union[str, bool]) -> bool: 

129 """Return a bool version of value. 

130 

131 This function will ensure that value is a bool, or that it is a string that can be interpreted 

132 as a bool. It will return a bool. If it cannot do that, it will raise ValueError. 

133 

134 Args: 

135 value: The value to be validated as a bool. 

136 Returns: 

137 The boolean interpretation of value. 

138 Raises: 

139 ValueError: If value cannot be interpreted as a boolean. 

140 """ 

141 if isinstance(value, str): 

142 # Recent versions of Pyramid define a settings.falsey, but version 1.5.6 does not so its 

143 # values have been backported here for the False case. Pyramid defines an asbool(), but it 

144 # will not raise any error for strings that aren't in the truthy or falsey lists, and we 

145 # want strict validation. 

146 if value.lower().strip() in settings.truthy: 

147 return True 

148 elif value.lower().strip() in ('f', 'false', 'n', 'no', 'off', '0'): 

149 return False 

150 else: 

151 raise ValueError('"{}" cannot be interpreted as a boolean value.'.format(value)) 

152 

153 if not isinstance(value, bool): 

154 raise ValueError('"{}" is not a bool or a string.'.format(value)) 

155 

156 return value

159def _validate_none_or(validator: typing.Callable[[typing.Any], typing.Any]) \ 

160 -> typing.Callable[[typing.Any], typing.Any]: 

161 """Return a function that will ensure a value is None or passes validator. 

162 

163 This function returns a function that will take a single argument, value, and will ensure 

164 that value is None or that it passes the given validator. 

165 

166 Args: 

167 validator: A function to apply when value is not None. 

168 Returns: 

169 A validator function that accepts an argument to be validated. 

170 """ 

171 def _validate(value: typing.Any) -> typing.Any: 

172 if value is None: 

173 return value 

174 

175 return validator(value) 

176 

177 return _validate

180def validate_path(value: str) -> str: 

181 """Ensure that value is an existing path on the local filesystem and return it. 

182 

183 Use os.path.exists to ensure that value is an existing path. Return the value if it is, else 

184 raise ValueError. 

185 

186 Args: 

187 value: The path to be validated. 

188 Returns: 

189 The path. 

190 Raises: 

191 ValueError: If os.path.exists returns False. 

192 """

194 raise ValueError(f'{value!r} does not exist.')

627 """Ensure the config is loaded, and then call the superclass copy.""" 

628 if not self.loaded: 

629 self.load_config()

632 def load_config(self, settings: typing.Mapping = None): 

633 """ 

634 Load the configuration either from the config file, or from the given settings. 

635 

636 args: 

637 settings: If given, the settings are pulled from this dictionary. Otherwise, the 

638 config file is used. 

639 """ 

640 self._load_defaults() 

641 configfile = get_configfile() 

642 if settings: 

643 self.update(settings) 

644 else: 

645 self.update(get_appsettings(configfile)) 

646 self.loaded = True 

647 self._validate()

649 def _load_defaults(self): 

650 """Iterate over self._defaults and set all default values on self.""" 

651 for k, v in self._defaults.items(): 

652 self[k] = v['value'] 

653 

654 def _validate(self): 

655 """Run the validators found in self._defaults on all the corresponding values.""" 

656 errors = [] 

657 for k in self._defaults.keys(): 

658 try: 

659 self[k] = self._defaults[k]['validator'](self[k]) 

660 except ValueError as e: 

661 errors.append('\t{}: {}'.format(k, str(e))) 

662 

663 if errors: 

664 raise ValueError( 

665 'Invalid config values were set: \n{}'.format('\n'.join(errors)))
