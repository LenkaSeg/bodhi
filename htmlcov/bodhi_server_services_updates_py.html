123def get_update(request): 

124 """ 

125 Return a single update from an id or alias. 

126 

127 Args: 

128 request (pyramid.request): The current request. 

129 Returns: 

130 dict: A dictionary with the following key mappings: 

131 update: The update that was requested. 

132 can_edit: A boolean indicating whether the update can be edited. 

133 """

185def set_request(request): 

186 """ 

187 Set a specific :class:`bodhi.server.models.UpdateRequest` on a given update. 

188 

189 Args: 

190 request (pyramid.request): The current request. 

191 Returns: 

192 dict: A dictionary mapping the key "update" to the update that was modified. 

193 """ 

194 update = request.validated['update'] 

195 action = request.validated['request'] 

196 

197 if update.locked: 

198 request.errors.add('body', 'request', 

199 "Can't change request on a locked update") 

200 return 

201 

202 if update.status == UpdateStatus.stable and action == UpdateRequest.testing: 

203 request.errors.add('body', 'request', 

204 "Pushing back to testing a stable update is not allowed") 

205 return 

206 

207 if action == UpdateRequest.stable: 

208 settings = request.registry.settings 

209 result, reason = update.check_requirements(request.db, settings) 

210 log.info( 

211 f'Unable to set request for {update.alias} to {action} due to failed requirements: ' 

212 f'{reason}') 

213 if not result: 

214 request.errors.add('body', 'request', 

215 'Requirement not met %s' % reason) 

216 return 

217 

218 try: 

219 update.set_request(request.db, action, request.user.name) 

220 except BodhiException as e: 

221 log.info("Failed to set the request: %s", e) 

222 request.errors.add('body', 'request', str(e)) 

223 except Exception as e: 

224 log.exception("Unhandled exception in set_request") 

225 request.errors.add('body', 'request', str(e)) 

226 

227 return dict(update=update)

259def query_updates(request): 

260 """ 

261 Search updates by given criteria. 

262 

263 Args: 

264 request (pyramid.request): The current request. 

265 Returns: 

266 dict: A dictionary with at least the following key mappings: 

267 updates: An iterable of the updates that match the query. 

268 page: The current page. 

269 pages: The total number of pages. 

270 rows_per_page: How many results on on the page. 

271 total: The total number of updates matching the query. 

272 package: The package corresponding to the first update found in the search. 

273 """ 

274 db = request.db 

275 data = request.validated 

276 query = db.query(Update) 

277 

278 approved_since = data.get('approved_since') 

279 if approved_since is not None: 

280 query = query.filter(Update.date_approved >= approved_since) 

281 

282 approved_before = data.get('approved_before') 

283 if approved_before is not None: 

284 query = query.filter(Update.date_approved < approved_before) 

285 

286 bugs = data.get('bugs') 

287 if bugs is not None: 

288 query = query.join(Update.bugs) 

289 query = query.filter(or_(*[Bug.bug_id == bug_id for bug_id in bugs])) 

290 

291 critpath = data.get('critpath') 

292 if critpath is not None: 

293 query = query.filter(Update.critpath == critpath) 

294 

295 like = data.get('like') 

296 if like is not None: 

297 query = query.join(Update.builds) 

298 query = query.filter(Build.nvr.like('%%%s%%' % like)) 

299 

300 search = data.get('search') 

301 if search is not None: 

302 query = query.join(Update.builds) 

303 query = query.filter(or_( 

304 Build.nvr.ilike('%%%s%%' % search), Update.alias.ilike('%%%s%%' % search))) 

305 

306 locked = data.get('locked') 

307 if locked is not None: 

308 query = query.filter(Update.locked == locked) 

309 

310 modified_since = data.get('modified_since') 

311 if modified_since is not None: 

312 query = query.filter(Update.date_modified >= modified_since) 

313 

314 modified_before = data.get('modified_before') 

315 if modified_before is not None: 

316 query = query.filter(Update.date_modified < modified_before) 

317 

318 packages = data.get('packages') 

319 if packages is not None: 

320 query = query.join(Update.builds).join(Build.package) 

321 query = query.filter(or_(*[Package.name == pkg for pkg in packages])) 

322 

323 package = None 

324 if packages and len(packages): 

325 package = packages[0] 

326 

327 builds = data.get('builds') 

328 if builds is not None: 

329 query = query.join(Update.builds) 

330 query = query.filter(or_(*[Build.nvr == build for build in builds])) 

331 

332 pushed = data.get('pushed') 

333 if pushed is not None: 

334 query = query.filter(Update.pushed == pushed) 

335 

336 pushed_since = data.get('pushed_since') 

337 if pushed_since is not None: 

338 query = query.filter(Update.date_pushed >= pushed_since) 

339 

340 pushed_before = data.get('pushed_before') 

341 if pushed_before is not None: 

342 query = query.filter(Update.date_pushed < pushed_before) 

343 

344 releases = data.get('releases') 

345 if releases is not None: 

346 query = query.filter(or_(*[Update.release == r for r in releases])) 

347 

348 # This singular version of the plural "releases" is purely for bodhi1 

349 # backwards compat (mostly for RSS feeds) - threebean 

350 release = data.get('release') 

351 if release is not None: 

352 query = query.filter(Update.release == release) 

353 

354 req = data.get('request') 

355 if req is not None: 

356 query = query.filter(Update.request == req) 

357 

358 severity = data.get('severity') 

359 if severity is not None: 

360 query = query.filter(Update.severity == severity) 

361 

362 status = data.get('status') 

363 if status is not None: 

364 query = query.filter(or_(*[Update.status == s for s in status])) 

365 

366 submitted_since = data.get('submitted_since') 

367 if submitted_since is not None: 

368 query = query.filter(Update.date_submitted >= submitted_since) 

369 

370 submitted_before = data.get('submitted_before') 

371 if submitted_before is not None: 

372 query = query.filter(Update.date_submitted < submitted_before) 

373 

374 suggest = data.get('suggest') 

375 if suggest is not None: 

376 query = query.filter(Update.suggest == suggest) 

377 

378 type = data.get('type') 

379 if type is not None: 

380 query = query.filter(Update.type == type) 

381 

382 content_type = data.get('content_type') 

383 if content_type is not None: 

384 query = query.join(Update.builds) 

385 query = query.filter(Build.type == content_type) 

386 

387 gating_status = data.get('gating') 

388 if gating_status is not None: 

389 query = query.filter(Update.test_gating_status == gating_status) 

390 

391 user = data.get('user') 

392 if user is not None: 

393 query = query.filter(or_(*[Update.user == u for u in user])) 

394 

395 updateid = data.get('updateid') 

396 if updateid is not None: 

397 query = query.filter(or_(*[Update.alias == uid for uid in updateid])) 

398 

399 alias = data.get('alias') 

400 if alias is not None: 

401 query = query.filter(or_(*[Update.alias == a for a in alias])) 

402 

403 from_side_tag = data.get('from_side_tag') 

404 if from_side_tag is not None: 

405 if from_side_tag: 

406 query = query.filter(Update.from_tag.isnot(None)) 

407 else: 

408 query = query.filter(Update.from_tag.is_(None)) 

409 

410 query = query.order_by(Update.date_submitted.desc()) 

411 

412 # We can't use ``query.count()`` here because it is naive with respect to 

413 # all the joins that we're doing above. 

414 count_query = query.with_labels().statement\ 

415 .with_only_columns([func.count(distinct(Update.id))])\ 

416 .order_by(None) 

417 total = db.execute(count_query).scalar() 

418 

419 page = data.get('page') 

420 rows_per_page = data.get('rows_per_page') 

421 pages = int(math.ceil(total / float(rows_per_page))) 

422 query = query.offset(rows_per_page * (page - 1)).limit(rows_per_page) 

423 

424 return_values = dict( 

425 updates=query.all(), 

426 page=page, 

427 pages=pages, 

428 rows_per_page=rows_per_page, 

429 total=total, 

430 chrome=data.get('chrome'), 

431 display_user=data.get('display_user', False), 

432 display_request=data.get('display_request', True), 

433 package=package, 

434 ) 

435 # we need some extra information for the searching / filterings interface 

436 # when rendering the html, so we add this here. 

437 if request.accept.accept_html(): 

438 return_values.update( 

439 gating_statuses=sorted( 

440 list(bodhi.server.models.TestGatingStatus.values())), 

441 types=list(bodhi.server.models.UpdateType.values()), 

442 severities=sorted( 

443 list(bodhi.server.models.UpdateSeverity.values()), 

444 key=bodhi.server.util.sort_severity), 

445 statuses=list(bodhi.server.models.UpdateStatus.values()), 

446 releases=Release.all_releases(), 

447 ) 

448 return return_values

468def new_update(request): 

469 """ 

470 Save an update. 

471 

472 This entails either creating a new update, or editing an existing one. To 

473 edit an existing update, the update's alias must be specified in 

474 the ``edited`` parameter. 

475 

476 If the ``from_tag`` parameter is specified and ``builds`` is missing or 

477 empty, the list of builds will be filled with the latest builds in this 

478 Koji tag. This is done by validate_from_tag() because the list of builds 

479 needs to be available in validate_acls(). 

480 

481 If the release is composed by Bodhi (i.e. a branched or stable release 

482 after the Bodhi activation point), ensure that related tags 

483 ``from_tag``-pending-signing and ``from_tag``-testing exists and if not 

484 create them in Koji. If the state of the release is not `pending`, add its 

485 pending-signing tag and remove it if it's a side tag. 

486 

487 Args: 

488 request (pyramid.request): The current request. 

489 """ 

490 data = request.validated 

491 log.debug('validated = %s' % data) 

492 

493 # This has already been validated at this point, but we need to ditch 

494 # it since the models don't care about a csrf argument. 

495 data.pop('csrf_token') 

496 

497 # Same here, but it can be missing. 

498 data.pop('builds_from_tag', None) 

499 data.pop('sidetag_owner', None) 

500 

501 build_nvrs = data.get('builds', []) 

502 from_tag = data.get('from_tag') 

503 

504 caveats = [] 

505 try: 

506 

507 releases = set() 

508 builds = [] 

509 

510 # Create the Package and Build entities 

511 for nvr in build_nvrs: 

512 name, version, release = request.buildinfo[nvr]['nvr'] 

513 

514 package = Package.get_or_create(request.db, request.buildinfo[nvr]) 

515 

516 # Also figure out the build type and create the build if absent. 

517 build_class = ContentType.infer_content_class( 

518 base=Build, build=request.buildinfo[nvr]['info']) 

519 build = build_class.get(nvr) 

520 

521 if build is None: 

522 log.debug("Adding nvr %s, type %r", nvr, build_class) 

523 build = build_class(nvr=nvr, package=package) 

524 request.db.add(build) 

525 request.db.flush() 

526 

527 build.package = package 

528 build.release = request.buildinfo[build.nvr]['release'] 

529 builds.append(build) 

530 releases.add(request.buildinfo[build.nvr]['release']) 

531 

532 # Disable manual updates for releases not composed by Bodhi 

533 # see #4058 

534 if not from_tag: 

535 for release in releases: 

536 if not release.composed_by_bodhi: 

537 request.errors.add('body', 'builds', 

538 "Cannot manually create updates for a Release which is not " 

539 "composed by Bodhi.\nRead the 'Automatic updates' page in " 

540 "Bodhi docs about this error.") 

541 request.db.rollback() 

542 return 

543 

544 # We want to go ahead and commit the transaction now so that the Builds are in the database. 

545 # Otherwise, there will be a race condition between robosignatory signing the Builds and the 

546 # signed handler attempting to mark the builds as signed. When we lose that race, the signed 

547 # handler doesn't see the Builds in the database and gives up. After that, nothing will mark 

548 # the builds as signed. 

549 request.db.commit() 

550 

551 # After we commit the transaction, we need to get the builds and releases again, 

552 # since they were tied to the previous session that has now been terminated. 

553 builds = [] 

554 releases = set() 

555 for nvr in build_nvrs: 

556 # At this moment, we are sure the builds are in the database (that is what the commit 

557 # was for actually). 

558 build = Build.get(nvr) 

559 builds.append(build) 

560 releases.add(build.release) 

561 

562 if data.get('edited'): 

563 

564 log.info('Editing update: %s' % data['edited']) 

565 

566 data['release'] = list(releases)[0] 

567 data['builds'] = [b.nvr for b in builds] 

568 data['from_tag'] = from_tag 

569 result, _caveats = Update.edit(request, data) 

570 caveats.extend(_caveats) 

571 else: 

572 if len(releases) > 1: 

573 caveats.append({ 

574 'name': 'releases', 

575 'description': 'Your update is being split ' 

576 'into %i, one for each release.' % len(releases) 

577 

578 }) 

579 updates = [] 

580 for release in releases: 

581 _data = copy.copy(data) # Copy it because .new(..) mutates it 

582 _data['builds'] = [b for b in builds if b.release == release] 

583 _data['release'] = release 

584 _data['from_tag'] = from_tag 

585 

586 log.info('Creating new update: %r' % _data['builds']) 

587 result, _caveats = Update.new(request, _data) 

588 log.debug('%s update created', result.alias) 

589 

590 updates.append(result) 

591 caveats.extend(_caveats) 

592 

593 if len(releases) > 1: 

594 result = dict(updates=updates) 

595 

596 if from_tag: 

597 for u in updates: 

598 builds = [b.nvr for b in u.builds] 

599 if not u.release.composed_by_bodhi: 

600 # Before the Bodhi activation point of a release, keep builds tagged 

601 # with the side-tag and its associate tags. 

602 side_tag_signing_pending = u.release.get_pending_signing_side_tag(from_tag) 

603 side_tag_testing_pending = u.release.get_pending_testing_side_tag(from_tag) 

604 handle_side_and_related_tags_task.delay( 

605 builds=builds, 

606 pending_signing_tag=side_tag_signing_pending, 

607 from_tag=from_tag, 

608 pending_testing_tag=side_tag_testing_pending) 

609 else: 

610 # After the Bodhi activation point of a release, add the pending-signing tag 

611 # of the release to funnel the builds back into a normal workflow for a 

612 # stable release. 

613 pending_signing_tag = u.release.pending_signing_tag 

614 candidate_tag = u.release.candidate_tag 

615 handle_side_and_related_tags_task.delay( 

616 builds=builds, 

617 pending_signing_tag=pending_signing_tag, 

618 from_tag=from_tag, 

619 candidate_tag=candidate_tag) 

620 

621 except LockedUpdateException as e: 

622 log.warning(str(e)) 

623 request.errors.add('body', 'builds', "%s" % str(e)) 

624 return 

625 except Exception as e: 

626 log.exception('Failed to create update') 

627 request.errors.add( 

628 'body', 'builds', 'Unable to create update. %s' % str(e)) 

629 return 

630 

631 # Obsolete older updates for three different cases... 

632 # editing an update, submitting a new single update, submitting multiple. 

633
