34def main(): 

35 """ 

36 Comment on updates that are eligible to be pushed to stable. 

37 

38 Queries for updates in the testing state that have a NULL request, and run approve_update on 

39 them. 

40 """ 

41 db_factory = transactional_session_maker() 

42 try: 

43 with db_factory() as db: 

44 testing = db.query(Update).filter_by(status=UpdateStatus.testing, request=None) 

45 for update in testing: 

46 approve_update(update, db) 

47 db.commit() 

48 except Exception: 

49 log.exception("There was an error approving testing updates.") 

50 finally: 

51 db_factory._end_session()

54def approve_update(update: Update, db: Session): 

55 """Add a comment to an update if it is ready for stable. 

56 

57 Check that the update is eligible to be pushed to stable but hasn't had comments from Bodhi to 

58 this effect. Add a comment stating that the update may now be pushed to stable. 

59 

60 Args: 

61 update: an update in testing that may be ready for stable. 

62 """ 

63 if not update.release.mandatory_days_in_testing and not update.autotime: 

64 # If this release does not have any testing requirements and is not autotime, 

65 # skip it 

66 log.info(f"{update.alias} doesn't have mandatory days in testing - bailing") 

67 return 

68 # If updates have reached the testing threshold, say something! Keep in mind 

69 # that we don't care about karma here, because autokarma updates get their request set 

70 # to stable by the Update.comment() workflow when they hit the required threshold. Thus, 

71 # this function only needs to consider the time requirements because these updates have 

72 # not reached the karma threshold. 

73 if not update.meets_testing_requirements: 

74 log.info(f"{update.alias} has not met testing requirements - bailing") 

75 return 

76 log.info(f'{update.alias} now meets testing requirements') 

77 # If the update is going to be pushed automatically to stable, do not 

78 # double comment that the maintainer can push it manually (#3846) 

79 if not update.autotime or update.days_in_testing < update.stable_days: 

80 # If this update was already commented, skip it 

81 if update.has_stable_comment: 

82 log.info(f"{update.alias} has already the comment that it can be pushed to stable - " 

83 "bailing") 

84 return 

85 # Only send email notification about the update reaching 

86 # testing approval on releases composed by bodhi 

87 update.comment( 

88 db, 

89 str(config.get('testing_approval_msg')), 

90 author='bodhi', 

91 email_notification=update.release.composed_by_bodhi 

92 ) 

93 notifications.publish(update_schemas.UpdateRequirementsMetStableV1.from_dict( 

94 dict(update=update))) 

95 if update.autotime and update.days_in_testing >= update.stable_days: 

96 if not update.has_stable_comment: 

97 notifications.publish(update_schemas.UpdateRequirementsMetStableV1.from_dict( 

98 dict(update=update))) 

99 log.info(f"Automatically marking {update.alias} as stable") 

100 # For now only rawhide update can be created using side tag 

101 # Do not add the release.pending_stable_tag if the update 

102 # was created from a side tag. 

103 if update.release.composed_by_bodhi: 

104 update.set_request(db=db, action=UpdateRequest.stable, username="bodhi") 

105 # For updates that are not included in composes run by bodhi itself, 

106 # mark them as stable 

107 else: 

108 # Single and Multi build update 

109 conflicting_builds = update.find_conflicting_builds() 

110 if conflicting_builds: 

111 builds_str = str.join(", ", conflicting_builds) 

112 update.comment( 

113 db, 

114 "This update cannot be pushed to stable. " 

115 f"These builds {builds_str} have a more recent " 

116 f"build in koji's {update.release.stable_tag} tag.", 

117 author="bodhi") 

118 update.request = None 

119 if update.from_tag is not None: 

120 update.status = UpdateStatus.pending 

121 update.remove_tag( 

122 update.release.get_pending_testing_side_tag(update.from_tag)) 

123 else: 

124 update.status = UpdateStatus.obsolete 

125 update.remove_tag(update.release.pending_testing_tag) 

126 update.remove_tag(update.release.candidate_tag) 

127 db.commit() 

128 log.info(f"{update.alias} has conflicting builds - bailing") 

129 return 

130 update.add_tag(update.release.stable_tag) 

131 update.status = UpdateStatus.stable 

132 update.request = None 

133 update.pushed = True 

134 update.date_stable = update.date_pushed = func.current_timestamp() 

135 update.comment(db, "This update has been submitted for stable by bodhi", 

136 author=u'bodhi') 

137 update.modify_bugs() 

138 db.commit() 

139 # Multi build update 

140 if update.from_tag: 

141 # Merging the side tag should happen here 

142 pending_signing_tag = update.release.get_pending_signing_side_tag( 

143 update.from_tag) 

144 testing_tag = update.release.get_pending_testing_side_tag(update.from_tag) 

145 update.remove_tag(pending_signing_tag) 

146 update.remove_tag(testing_tag) 

147 update.remove_tag(update.from_tag) 

148 # Delete side-tag and its children after Update has enter stable 

149 # We can't fully rely on Koji's auto-purge-when-empty because 

150 # there may be older nvrs tagged in the side-tag 

151 koji = buildsys.get_session() 

152 koji.multicall = True 

153 koji.deleteTag(pending_signing_tag) 

154 koji.deleteTag(testing_tag) 

155 koji.deleteTag(update.from_tag) 

156 koji.multiCall() 

157 else: 

158 # Single build update 

159 update.remove_tag(update.release.pending_testing_tag) 

160 update.remove_tag(update.release.pending_stable_tag) 

161 update.remove_tag(update.release.pending_signing_tag) 

162 update.remove_tag(update.release.candidate_tag) 

163 

164 log.info(f'{update.alias} processed by approve_testing')
